```bash
# run.sh — Invoke FunLexYacc via the LangThree interpreter
#
# Usage:
#   ./run.sh [args...]
#
# FunLexYacc has NO .NET build. All .fun files are interpreted by LangThree.
# To run the main entry point:
#   ./run.sh
#
# The LangThree binary path is controlled by LANGTHREE_BIN (default: Release build).

LANGTHREE_BIN="${LANGTHREE_BIN:-/Users/ohama/vibe-coding/LangThree/src/LangThree/bin/Release/net10.0/LangThree}"

    echo "ERROR: LangThree binary not found at $LANGTHREE_BIN"
    echo "Set LANGTHREE_BIN to the correct path or build LangThree first."
    exit 1
fi

exec "$LANGTHREE_BIN" "$(dirname "$0")/src/Main.fun" "$@"
```
