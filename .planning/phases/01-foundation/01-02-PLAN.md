  <action>
Create tests/harness/compare.sh (chmod +x). The script compares a reference binary against a candidate binary on a single input and checks both token output and AST output.

Interface:
  compare.sh <ref-binary> <cand-binary> <input-file>
  compare.sh <ref-binary> <cand-binary> --expr "<expression>"

Exit codes: 0 = both tokens AND AST match, 1 = any difference or error.

Full script:
```bash
#!/usr/bin/env bash
# compare.sh — side-by-side comparison: reference vs candidate
#
# Usage (file mode):
#   compare.sh <ref-binary> <cand-binary> <input-file>
#
# Usage (expr mode):
#   compare.sh <ref-binary> <cand-binary> --expr "<expression>"
#
# Exit codes: 0 = PASS (tokens and AST identical), 1 = FAIL
#
# Override reference binary via LANGTHREE_BIN env var.
# This script checks BOTH --emit-tokens and --emit-ast output.

set -euo pipefail

REF="${1:?Usage: compare.sh <ref-binary> <cand-binary> <input-file|--expr expr>}"
CAND="${2:?Usage: compare.sh <ref-binary> <cand-binary> <input-file|--expr expr>}"
shift 2

# Validate binaries exist
if [ ! -x "$REF" ]; then
    echo "ERROR: Reference binary not found or not executable: $REF"
    exit 1
fi
if [ ! -x "$CAND" ]; then
    echo "ERROR: Candidate binary not found or not executable: $CAND"
    exit 1
fi

# Build the args array for both invocations
PASS=true

check_mode() {
    local mode="$1"      # tokens or ast
    local flag="$2"      # --emit-tokens or --emit-ast
    shift 2
    local args=("$@")

    local ref_out cand_out
    ref_out=$("$REF" "$flag" "${args[@]}" 2>/dev/null) || true
    cand_out=$("$CAND" "$flag" "${args[@]}" 2>/dev/null) || true

    if [ "$ref_out" = "$cand_out" ]; then
        echo "PASS ($mode)"
    else
        echo "FAIL ($mode)"
        diff <(echo "$ref_out") <(echo "$cand_out") | head -40
        PASS=false
    fi
}

check_mode "tokens" "--emit-tokens" "$@"
check_mode "ast"    "--emit-ast"    "$@"

if $PASS; then
    exit 0
else
    exit 1
fi
```

IMPORTANT: --expr mode is passed through naturally via "$@" — if the caller does `compare.sh ref cand --expr "1+1"`, the args array becomes `["--expr", "1+1"]` and both `"$REF" --emit-tokens --expr "1+1"` and `"$CAND" --emit-tokens --expr "1+1"` are invoked correctly. No special-casing needed.

Do NOT hardcode any binary paths in the script itself. Both ref and cand come from arguments.

Command to run the batch smoke-test:
```bash
REF=/Users/ohama/vibe-coding/LangThree/src/LangThree/bin/Release/net10.0/LangThree
PASS_COUNT=0
FAIL_COUNT=0

for flt in $(ls tests/fixtures/*.flt 2>/dev/null | head -10); do
    # Extract input section between "--- Input:" and next "---" marker
    INPUT_TEXT=$(awk '/^\/\/ --- Input:/{found=1; next} /^\/\/ ---/{found=0} found && !/^\/\//{print}' "$flt")
    if [ -z "$INPUT_TEXT" ]; then
        continue  # skip tests with no Input section (--expr tests)
    fi
    TMPFILE=$(mktemp /tmp/flt-input-XXXXX.fun)
    echo "$INPUT_TEXT" > "$TMPFILE"
    if tests/harness/compare.sh "$REF" "$REF" "$TMPFILE" > /dev/null 2>&1; then
        PASS_COUNT=$((PASS_COUNT + 1))
    else
        FAIL_COUNT=$((FAIL_COUNT + 1))
        echo "UNEXPECTED FAIL: $flt"
    fi
    rm -f "$TMPFILE"
done

echo "Smoke test: $PASS_COUNT passed, $FAIL_COUNT failed (all should pass — ref vs ref)"
```
